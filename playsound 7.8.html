    Simple example: send data to PHP and show the return results :

$("document").ready(function(){
    $(".form-data").submit(function(){
  
      data = $(this).serialize();
      $.ajax({
        type: "POST",
        dataType: "json",
        url: "example.php", //Relative or absolute path to response.php file
        data: data,
        success: function(data) {
          if(data.type === 'error')
          {
              
                $.alert({
                icon: 'flaticon-alarm-1',
                title: data.issue,
                message: data.opinion                          
                },{
                type: 'danger'
                });
                $.playSound("images/beeb.mp3");
               

          }else if(data.type === 'ok')
          {
                $.alert({
                icon: 'flaticon-alarm-1',
                title: data.thanks,
                message: data.welcome                         
                },{
                type: 'success'
                });
                $.playSound("images/beeb.mp3");
          }
  
        }
      });
      return false;
    });
  });
